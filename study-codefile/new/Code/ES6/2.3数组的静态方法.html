Array.isArray()
用于判断是否为数组

用法:
let arr = [1, 2, 3]
let res = Array.isArray(arr)
// res = true

Array.of()
用于将一组值, 转换为数组

用法:
let arr = Array.of(1, 2, 3)
// arr = [1, 2, 3]

Array.from()
用于将类数组对象, 转换为数组

用法:
let arr = Array.from('hello')
// arr = ['h', 'e', 'l', 'l', 'o']

let arr = Array.from({length: 3})
// arr = [undefined, undefined, undefined]

let arr = Array.from({length: 3}, () => 1)
// arr = [1, 1, 1]

let arr = Array.from({length: 3}, (item, index) => index)
// arr = [0, 1, 2]
例如获取标签
let p = document.querySelectorAll('p')
let arr = Array.from(p)

Array.from()
第二个参数, 用于对数组中的每一项进行处理, 类似数组的map方法

用法:
let arr = Array.from([1, 2, 3], (item) => item * 2)
// arr = [2, 4, 6]

map()
用于对数组中的每一项进行处理, 返回一个新数组

用法:
let arr = [1, 2, 3].map((item) => item * 2)
// arr = [2, 4, 6]

filter()
用于对数组中的每一项进行过滤, 返回一个新数组

用法:
let arr = [1, 2, 3].filter((item) => item > 1)
// arr = [2, 3]

find()
用于对数组中的每一项进行查找, 返回第一个符合条件的值

用法:
let arr = [1, 2, 3].find((item) => item > 1)
// arr = 2

includes()
用于判断数组中是否包含某个值, 返回一个布尔值
参数1: 要查找的值
参数2: 查找的起始位置(可选)

用法:
let arr = [1, 2, 3].includes(2)
// arr = true

indexOf()
用于判断数组中是否包含某个值, 返回一个索引值
索引从0开始, 如果没有找到, 返回-1
参数1: 要查找的值
参数2: 查找的起始位置(可选)

用法:
let arr = [1, 2, 3].indexOf(2)
// arr = 1

includes()和indexOf()的区别

includes()返回布尔值
indexOf()返回索引值 indexOf() 还返回隐形布尔值, -1为false, 其他为true, 通过判断是否为-1, 来返回布尔值
indexOf()判断例:
let condition = [1, 2, 3].indexOf(2) !== -1
console.log(condition) // true

includes()可以判断NaN, indexOf()不可以判断NaN

fill()
用于填充数组, 返回一个新数组
参数1: 填充的值
参数2: 填充的起始位置(可选)
参数3: 填充的结束位置(可选)

用法:
let arr = [1, 2, 3].fill(0)
// arr = [0, 0, 0]

let arr = [1, 2, 3].fill(0, 1, 2)
// arr = [1, 0, 3]

let arr = [1, 2, 3].fill(0, 1)
// arr = [1, 0, 0]

copyWithin()
用于复制数组中的一部分, 替换到数组中的另一部分, 返回一个新数组
参数1: 从该位置开始替换数据, 如果为负数, 则从数组末尾开始计算
参数2(可选): 从该位置开始读取数据, 默认为0, 如果为负数, 则从数组末尾开始计算
参数3(可选): 到该位置前停止读取数据, 默认为数组长度

用法:
let arr = [1, 2, 3, 4, 5].copyWithin(0, 3)
// 从索引3开始复制, 复制到数组末尾, 然后替换到索引0开始的位置
// arr = [4, 5, 3, 4, 5]

let arr = [1, 2, 3, 4, 5].copyWithin(0, 3, 4)
// 从索引3开始复制, 复制到索引4前停止, 然后替换到索引0开始的位置
// arr = [4, 2, 3, 4, 5]
let arr = [1, 2, 3, 4, 5].copyWithin(0, -2, -1)
// 从索引-2开始复制, 复制到索引-1前停止, 然后替换到索引0开始的位置
// arr = [4, 2, 3, 4, 5]

some()
用于判断数组中是否有一项符合条件, 返回一个布尔值

用法:
let arr = [1, 2, 3].some((item) => item > 1)
// arr = true

every()
用于判断数组中是否所有项都符合条件, 返回一个布尔值

用法:
let arr = [1, 2, 3].every((item) => item > 1)
// arr = false

Object.values()
用于获取对象中的所有值, 返回一个数组

用法:
let obj = {name: 'yuki', age: 18}
let arr = Object.values(obj)
// arr = ['yuki', 18]

entries()
用于获取对象中的所有键值对, 返回一个数组

用法:
let obj = {name: 'yuki', age: 18}
let arr = Object.entries(obj)
// arr = [['name', 'yuki'], ['age', 18]]

keys()
用于获取对象中的所有键名, 返回一个数组

用法:
let obj = {name: 'yuki', age: 18}
let arr = Object.keys(obj)
// arr = ['name', 'age']

fromEntries()
用于将键值对转换为对象, 返回一个对象

用法:
let arr = [['name', 'yuki'], ['age', 18]]
let obj = Object.fromEntries(arr)
// obj = {name: 'yuki', age: 18}

values()
用于获取对象中的所有值, 返回一个数组

用法:
let obj = {name: 'yuki', age: 18}
let arr = Object.values(obj)
// arr = ['yuki', 18]

find()
用于对数组中的每一项进行查找, 返回第一个符合条件的值
参数是一个回调函数, 回调函数中有三个参数
参数1: 当前项的值
参数2: 当前项的索引
参数3: 当前数组

用法:
let arr = [1, 2, 3].find((item) => item > 1)
// arr = 2
三个参数的用法:
let arr = [1, 2, 3].find((item, index, arr) => {
console.log(item)
console.log(index)
console.log(arr)
})
