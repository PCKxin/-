解构赋值:
用于从数组或者对象中提取值, 对变量进行赋值, 称为解构赋值
解构赋值的本质是右边的值, 赋值给左边的变量
解构的源是右边的值, 解构的目标是左边的变量
解构赋值的语法:

普通:
let [a, b, c] = [1, 2, 3]
// a = 1, b = 2, c = 3
let {a, b, c} = {a: 1, b: 2, c: 3}
// a = 1, b = 2, c = 3

嵌套:
let [a, [b, c], d] = [1, [2, 3], 4]
// a = 1, b = 2, c = 3, d = 4
let {a, b: {c, d}} = {a: 1, b: {c: 2, d: 3}}
// a = 1, c = 2, d = 3

默认值:
匹配结果为undefined时, 变量的默认值才会生效
let [a, b = 2] = [1]
// a = 1, b = 2
let {a, b = 2} = {a: 1}
// a = 1, b = 2
let [a, b = 2] = [1, undefined]
// a = 1, b = 2
let {a, b = 2} = {a: 1, b: undefined}
// a = 1, b = 2

忽略:
let [a, , b] = [1, 2, 3]
// a = 1, b = 3
let {a, , b} = {a: 1, b: 2, c: 3}
// a = 1, b = 3

交换:
var a = 1, b = 2
var [a, b] = [b, a]
// a = 2, b = 1
var a = 1, b = 2
var temp = a
var a = b
var b = temp
// a = 2, b = 1

不完全解构:
let [a, [b], c] = [1, [2, 3], 4]
// a = 1, b = 2, c = 4
let [a, [b], c] = [1, [2, 3]]
// a = 1, b = 2, c = undefined

剩余运算符:
let [a, ...b] = [1, 2, 3]
// a = 1, b = [2, 3]
let {a, ...b} = {a: 1, b: 2, c: 3}
// a = 1, b = {b: 2, c: 3}

解构默认值
let {a = 10, b = 5} = {a: 3}; // a = 3; b = 5; let {a: aa = 10, b: bb = 5} = {a: 3}; // aa = 3; bb = 5;
