模版字符串:

${}
用于拼接字符串
可以搭配``使用
用法:
let a = 1
let b = 2
let c = `${a} + ${b} = ${a + b}`
函数:
function fn(){
return 'Hello World'
}
// es6 函数写法
let fn = () => 'Hello World'
let c = `${fn()}`
函数带参数
function fn(a, b){
return `${a} + ${b} = ${a + b}`
}
// es6 函数写法
let fn = (a, b) => `${a} + ${b} = ${a + b}`
let c = `${fn(1, 2)}`
