es6 允许函数设置参数默认值
通常情况下, 设置默认值的参数要靠后
非尾部的参数设置默认值, 不能省略参数, 否则会报错
省略参数报错例:
function fn(x,y=1,z){
return [x,y,z];
}
console.log(fn(1,,2)); // 报错

用法:
function fn(a,b=1){
return a+b;
}
console.log(fn(1)); // 2
console.log(fn(1,2)); // 3
console.log(fn(1,undefined)); // 2
console.log(fn(1,null)); // 1
console.log(fn(1,NaN)); // NaN
console.log(fn(1,0)); // 1
